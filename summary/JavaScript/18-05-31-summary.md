- while문은 무한루프를 돌리기에 좋다 ex) 게임에서 접속해 있는 한 계속 이동하는 부분
```
var n = 0;			// 초기화문 while문 앞에 
var x = 0;

while (n < 3){		// n이 3보다 작을 때까지 계속 반복한다
n++;				// 증감문
x += n;
console.log(n);
}
```
